
  it("should return 400 for title exceeding maximum length", async () => {
    const longTitle = "a".repeat(101);
    const request = new Request("http://localhost/api/notes", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ title: longTitle, content: "Test content" }),
    });
    const response = await POST(request);
    const data = await response.json();
    expect(response.status).toBe(400);
    expect(data.errors).toHaveLength(1);
    expect(data.errors[0].path).toBe("title");
    expect(data.errors[0].message).toContain("100 characters");
  });
